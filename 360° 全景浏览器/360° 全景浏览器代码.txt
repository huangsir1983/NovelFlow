    <script>
        // --- UI 元素 ---
        const dropZone = document.getElementById('drop-zone');
        const fileInput = document.getElementById('file-input');
        const uploadOverlay = document.getElementById('upload-overlay');
        const loadingOverlay = document.getElementById('loading-overlay');
        const loadingText = document.getElementById('loading-text');
        const controlsPanel = document.getElementById('controls-panel');
        const btnChangeImage = document.getElementById('btn-change-image');
        const btnRecenter = document.getElementById('btn-recenter');
        const btnScreenshot = document.getElementById('btn-screenshot');

        // --- Three.js 全局变量 ---
        let scene, camera, renderer, controls, sphereMaterial;

        // --- 初始化 Three.js 场景 ---
        function initThreeJS() {
            const container = document.getElementById('canvas-container');

            // 1. 创建场景
            scene = new THREE.Scene();

            // 2. 创建相机 (视角75度, 比例, 近裁剪面, 远裁剪面)
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(0, 0, 0.1); // 放置在原点附近

            // 3. 创建渲染器
            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true, preserveDrawingBuffer: true });
            renderer.setPixelRatio(window.devicePixelRatio);
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.outputEncoding = THREE.sRGBEncoding; // 确保颜色显示正常
            container.appendChild(renderer.domElement);

            // 4. 创建轨道控制器 (允许用户通过鼠标拖拽环顾四周)
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableZoom = true;
            controls.enablePan = false;    // 禁用平移，保持在中心
            controls.minDistance = 0.1;
            controls.maxDistance = 100;    // 限制缩放范围在球体内
            controls.rotateSpeed = -0.5;   // 负数使拖拽方向符合内视角直觉

            // 5. 创建全景球体
            // 使用较大的半径(500)，较高的分段数以保证图像平滑
            const geometry = new THREE.SphereGeometry(500, 60, 40);
            
            // 反转球体的面，使我们在球体内部能看到贴图
            geometry.scale(-1, 1, 1);

            // 初始材质为空（深色背景）
            sphereMaterial = new THREE.MeshBasicMaterial({ color: 0x111111 });
            const sphere = new THREE.Mesh(geometry, sphereMaterial);
            scene.add(sphere);

            // 监听窗口大小变化
            window.addEventListener('resize', onWindowResize, false);

            // 开始动画循环
            animate();
        }

        // --- 动画循环 ---
        function animate() {
            requestAnimationFrame(animate);
            controls.update(); // 必须在动画循环中更新控制器
            renderer.render(scene, camera);
        }

        // --- 窗口大小自适应 ---
        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        // --- 加载全景图片 ---
        function loadPanorama(imageDataUrl) {
            // 显示加载动画
            loadingOverlay.classList.remove('hidden');

            const textureLoader = new THREE.TextureLoader();
            
            textureLoader.load(
                imageDataUrl,
                // 加载成功的回调
                function (texture) {
                    texture.encoding = THREE.sRGBEncoding;
                    
                    // 为了防止接缝处的伪影，通常全景图不需要 mipmap，或者需要特殊设置
                    texture.minFilter = THREE.LinearFilter;
                    texture.magFilter = THREE.LinearFilter;

                    // 应用贴图并恢复材质颜色为纯白（以显示原始图片颜色）
                    sphereMaterial.map = texture;
                    sphereMaterial.color.setHex(0xffffff);
                    sphereMaterial.needsUpdate = true;

                    // 隐藏加载动画和上传层，显示控制栏
                    loadingOverlay.classList.add('hidden');
                    uploadOverlay.classList.add('hidden');
                    controlsPanel.classList.remove('hidden');
                    
                    // 重置相机视角
                    recenterCamera();
                },
                // 加载进度的回调
                undefined,
                // 加载失败的回调
                function (err) {
                    console.error('加载贴图出错:', err);
                    loadingOverlay.classList.add('hidden');
                    alert('图片加载失败，请尝试其他图片。');
                }
            );
        }

        // --- 处理文件读取 ---
        function handleFile(file) {
            if (!file) return;
            
            // 验证是否为图片
            if (!file.type.startsWith('image/')) {
                alert('请上传图片文件 (JPG, PNG, WEBP)。');
                return;
            }

            const reader = new FileReader();
            
            // 显示加载提示
            loadingOverlay.classList.remove('hidden');
            
            reader.onload = function (e) {
                // 读取完毕后将 dataURL 交给 Three.js 处理
                loadPanorama(e.target.result);
            };
            
            reader.onerror = function () {
                loadingOverlay.classList.add('hidden');
                alert('读取文件出错！');
            };

            // 以 Data URL 的形式读取文件
            reader.readAsDataURL(file);
        }

        // --- 交互事件绑定 ---
        
        // 点击上传区域触发文件选择
        dropZone.addEventListener('click', () => {
            fileInput.click();
        });

        btnChangeImage.addEventListener('click', () => {
            fileInput.click();
        });

        // 重置相机视角
        function recenterCamera() {
            camera.position.set(0, 0, 0.1);
            controls.target.set(0, 0, 0);
            camera.fov = 75;
            camera.updateProjectionMatrix();
            controls.update();
        }
        btnRecenter.addEventListener('click', recenterCamera);

        // 生成四宫格截图
        async function takeFourGridScreenshot() {
            // 显示加载提示
            loadingText.innerText = '正在生成四宫格截图...';
            loadingOverlay.classList.remove('hidden');

            // 稍微延迟以确保 UI 渲染出加载动画
            await new Promise(resolve => setTimeout(resolve, 50));

            // 记录当前渲染器和相机状态
            const origWidth = window.innerWidth;
            const origHeight = window.innerHeight;
            const origAspect = camera.aspect;
            const origPos = camera.position.clone();
            const origTarget = controls.target.clone();

            // 设定输出的最终 16:9 尺寸 (1920x1080)
            const finalWidth = 1920;
            const finalHeight = 1080;
            const cellW = finalWidth / 2; // 960
            const cellH = finalHeight / 2; // 540

            // 创建离屏 Canvas 用于拼接
            const canvasObj = document.createElement('canvas');
            canvasObj.width = finalWidth;
            canvasObj.height = finalHeight;
            const ctx = canvasObj.getContext('2d');

            // 调整渲染器和相机以匹配单格 16:9 比例
            renderer.setSize(cellW, cellH);
            camera.aspect = cellW / cellH;
            camera.updateProjectionMatrix();

            // 暂时禁用控制器防止干扰相机转向
            controls.enabled = false;

            // 定义四个视角的绝对方向 (前, 右, 左, 后)
            const directions = [
                { x: 0, z: -1 }, // 前 (映射到左上)
                { x: 1, z: 0 },  // 右 (映射到右上)
                { x: -1, z: 0 }, // 左 (映射到左下)
                { x: 0, z: 1 }   // 后 (映射到右下)
            ];

            const images = [];

            // 渲染四个视角的画面并提取为 Base64
            for (let i = 0; i < directions.length; i++) {
                camera.position.set(0, 0, 0);
                camera.lookAt(directions[i].x, 0, directions[i].z);
                
                renderer.render(scene, camera);
                const dataURL = renderer.domElement.toDataURL('image/jpeg', 0.95);
                images.push(dataURL);
            }

            // 恢复原始页面渲染状态
            renderer.setSize(origWidth, origHeight);
            camera.aspect = origAspect;
            camera.position.copy(origPos);
            controls.target.copy(origTarget);
            camera.updateProjectionMatrix();
            controls.enabled = true;
            controls.update();

            // 将截取的 Base64 转换为图片对象以绘制到 Canvas
            const loadedImages = await Promise.all(images.map(src => {
                return new Promise(resolve => {
                    const img = new Image();
                    img.onload = () => resolve(img);
                    img.src = src;
                });
            }));

            // 绘制 2x2 四宫格
            ctx.drawImage(loadedImages[0], 0, 0, cellW, cellH);           // 左上: 前
            ctx.drawImage(loadedImages[1], cellW, 0, cellW, cellH);       // 右上: 右
            ctx.drawImage(loadedImages[2], 0, cellH, cellW, cellH);       // 左下: 左
            ctx.drawImage(loadedImages[3], cellW, cellH, cellW, cellH);   // 右下: 后

            // 绘制白色十字边框以清晰分隔四宫格
            ctx.strokeStyle = '#ffffff';
            ctx.lineWidth = 4;
            ctx.beginPath();
            ctx.moveTo(cellW, 0); ctx.lineTo(cellW, finalHeight);
            ctx.moveTo(0, cellH); ctx.lineTo(finalWidth, cellH);
            ctx.stroke();

            // 导出最终大图并触发下载
            const finalDataURL = canvasObj.toDataURL('image/jpeg', 0.95);
            const link = document.createElement('a');
            link.download = 'panorama-4grid.jpg';
            link.href = finalDataURL;
            link.click();

            // 恢复 UI 状态
            loadingText.innerText = '正在处理全景图...';
            loadingOverlay.classList.add('hidden');
        }
        
        btnScreenshot.addEventListener('click', takeFourGridScreenshot);

        // 文件输入框值改变时触发
        fileInput.addEventListener('change', (e) => {
            const file = e.target.files[0];
            handleFile(file);
            // 清空 input 值，允许重复上传同一张图片
            e.target.value = '';
        });

        // 拖拽事件支持
        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
            window.addEventListener(eventName, preventDefaults, false);
        });

        function preventDefaults(e) {
            e.preventDefault();
            e.stopPropagation();
        }

        // 拖拽视觉反馈
        ['dragenter', 'dragover'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => {
                dropZone.classList.add('drag-active');
            }, false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => {
                dropZone.classList.remove('drag-active');
            }, false);
        });

        // 处理放下事件
        window.addEventListener('drop', (e) => {
            const dt = e.dataTransfer;
            const files = dt.files;
            
            if (files.length > 0) {
                handleFile(files[0]);
            }
        }, false);

        // --- 页面加载完成执行初始化 ---
        window.onload = initThreeJS;
    </script>
</body>
